!/bin/bash
  echo "Begin"
  echo"*******************************"
  for password in `cat $1`;
  do       
      outguess -r flag.jpg -k $password -t flag.txt
      grep "hgame" flag.txt > /dev/null
      if [[ $? -eq 0 ]];
     then

     ┊   fi
     done
